decisione(fold). decisione(call). decisione(raise).

scelta(X)|notScelta(X):- decisione(X).

%solo una scelta per volta
:- not #count{X:scelta(X)}=1.

%non e possibile che la scelta non sia fold se la puntata minima è maggiore del budget.
:- notScelta(fold),puntataMinima(X),budget(Y),X>Y.

%non e possibile che la scelta  sia fold  se la puntata minima e 0
:-scelta(fold),puntataMinima(X),X=0.

%non e possibile che la scelta non sia fold se più di un avversario fa raise e la probabilità di vittoria e minore del 80
:- notScelta(fold),#count{X:sceltaAvversario(X,raise)}>1,probabilitaVittoria(Z),Z<=80.

%non e possibile che la scelta sia raise se puntata minima=Budget
:-scelta(raise),puntataMinima(X),budget(X).

%non e possibile che la scelta non sia raise se tutti hanno fatto call e le probabilità vittoria e maggiore del 90
:-notScelta(raise),avversari(N),#count{X:sceltaAvversario(X,call)}=N,probabilitaVittoria(Z),Z>90.
